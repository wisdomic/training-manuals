8.1 Gemini API คืออะไร

Gemini API คือช่องทางให้โปรแกรมของเรา "คุยกับ" Gemini AI ได้โดยตรง ผ่าน HTTP Request ทำให้สร้างแอปพลิเคชันที่มี AI ได้โดยไม่ต้องสร้าง model เอง

📡 REST API ทำงานอย่างไร
┌──────────────┐         HTTP POST          ┌──────────────────┐
│              │  ─────────────────────────▶ │                  │
│  แอปของเรา    │     + API Key + Prompt      │  Google Gemini   │
│  (GAS/Python) │  ◀───────────────────────── │  Server          │
│              │      JSON Response          │                  │
└──────────────┘                             └──────────────────┘
Endpoint หลัก:
POST https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=YOUR_API_KEY

8.2 ใช้ Gemini API กับ Google Apps Script

Google Apps Script (GAS) เป็นภาษา JavaScript ที่ทำงานบน Google Cloud เหมาะสำหรับสร้างแอปที่เชื่อมต่อกับ Google Workspace

📋 ขั้นตอนเตรียมตัว
  1. เข้า https://aistudio.google.com → สร้าง API Key (บทที่ 7)
  2. เปิด Google Sheets ใหม่ → ไปที่ Extensions → Apps Script
  3. เก็บ API Key ใน Script Properties:
    • ไปที่ Project Settings (ฟันเฟือง) → Script Properties
    • เพิ่ม: Key = GEMINI_API_KEY, Value = AIzaSy...
💻 โค้ดพื้นฐาน: เรียก Gemini API จาก GAS
/**
 * เรียก Gemini API จาก Google Apps Script
 * @param {string} prompt - คำถามหรือคำสั่ง
 * @returns {string} คำตอบจาก Gemini
 */
function callGemini(prompt) {
  var apiKey = PropertiesService.getScriptProperties().getProperty('GEMINI_API_KEY');
  var url = 'https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=' + apiKey;
  
  var payload = {
    contents: [{
      parts: [{
        text: prompt
      }]
    }],
    generationConfig: {
      temperature: 0.7,
      maxOutputTokens: 2048
    }
  };
  
  var options = {
    method: 'post',
    contentType: 'application/json',
    payload: JSON.stringify(payload),
    muteHttpExceptions: true
  };
  
  try {
    var response = UrlFetchApp.fetch(url, options);
    var json = JSON.parse(response.getContentText());
    
    if (json.candidates && json.candidates[0]) {
      return json.candidates[0].content.parts[0].text;
    } else {
      return 'Error: ไม่ได้รับคำตอบ';
    }
  } catch (e) {
    return 'Error: ' + e.toString();
  }
}

 * ทดสอบเรียก Gemini
 */

  var result = callGemini('สวัสดี คุณเป็น AI อะไร? ตอบสั้นๆ 1 ประโยค');
  Logger.log(result);

💻 โค้ดขั้นสูง: เรียกพร้อม System Instruction
/**
 * เรียก Gemini API พร้อม System Instruction
 * @param {string} systemInstruction - คำสั่งระบบ (บุคลิก/กฎ)
 * @param {string} userPrompt - คำถามจากผู้ใช้
 * @returns {string} คำตอบจาก Gemini
 */
function callGeminiWithSystem(systemInstruction, userPrompt) {
  var apiKey = PropertiesService.getScriptProperties().getProperty('GEMINI_API_KEY');
  var url = 'https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=' + apiKey;
  
  var payload = {
    system_instruction: {
      parts: [{
        text: systemInstruction
      }]
    },
    contents: [{
      role: 'user',
      parts: [{
        text: userPrompt
      }]
    }],
    generationConfig: {
      temperature: 0.5,
      maxOutputTokens: 4096
    }
  };
  
  var options = {
    method: 'post',
    contentType: 'application/json',
    payload: JSON.stringify(payload),
    muteHttpExceptions: true
  };
  
  var response = UrlFetchApp.fetch(url, options);
  var json = JSON.parse(response.getContentText());
  
  if (json.candidates && json.candidates[0]) {
    return json.candidates[0].content.parts[0].text;
  }
  return 'ไม่สามารถประมวลผลได้';
}

📊 สร้าง Custom Function ใน Sheets

สร้างฟังก์ชันที่เรียกใช้ใน cell ของ Google Sheets ได้เลย!

/**
 * Custom Function: ใช้ Gemini ตอบคำถามจาก cell
 * ใช้ใน Sheets: =GEMINI("สรุปข้อความนี้")
 * @param {string} prompt คำถามหรือคำสั่ง
 * @return {string} คำตอบจาก Gemini
 * @customfunction
 */
function GEMINI(prompt) {
  if (!prompt) return '';
  return callGemini(prompt);
}

/**
 * Custom Function: แปลภาษา
 * ใช้ใน Sheets: =GEMINI_TRANSLATE(A2, "อังกฤษ")
 * @param {string} text ข้อความ
 * @param {string} targetLang ภาษาปลายทาง
 * @return {string} ข้อความที่แปลแล้ว
 * @customfunction
 */
function GEMINI_TRANSLATE(text, targetLang) {
  if (!text) return '';
  var prompt = 'แปลข้อความต่อไปนี้เป็นภาษา' + targetLang + ' ตอบเฉพาะคำแปล: ' + text;
  return callGemini(prompt);
}

/**
 * Custom Function: สรุปข้อความ
 * ใช้ใน Sheets: =GEMINI_SUMMARIZE(A2, 3)
 * @param {string} text ข้อความ
 * @param {number} sentences จำนวนประโยค
 * @return {string} ข้อความที่สรุป
 * @customfunction
 */
function GEMINI_SUMMARIZE(text, sentences) {
  if (!text) return '';
  sentences = sentences || 3;
  var prompt = 'สรุปข้อความต่อไปนี้เป็น ' + sentences + ' ประโยค ภาษาไทย: ' + text;
  return callGemini(prompt);
}
⚠️ ข้อจำกัด Custom Function:
  • Custom function ใน Sheets มี timeout 30 วินาที
  • ไม่สามารถเรียกซ้ำเร็วๆ ได้ (Rate limit)
  • ถ้ามีหลาย cell ที่ใช้ =GEMINI() อาจโดน quota limit
  • แนะนำ: ใช้กับข้อมูลทีละไม่เกิน 10-20 cell

/**
 * AI Chatbot — ตอบคำถามจากข้อมูลใน Sheet FAQ
 */
function chatbotAnswer(userQuestion) {
  // 1. ดึงข้อมูล FAQ จาก Sheet
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var faqSheet = ss.getSheetByName('FAQ');
  var data = faqSheet.getDataRange().getValues();
  
  // 2. สร้าง context จากข้อมูล FAQ
  var faqContext = '';
  for (var i = 1; i < data.length; i++) {
    faqContext += 'หมวด: ' + data[i][0] + '\n';
    faqContext += 'ถาม: ' + data[i][1] + '\n';
    faqContext += 'ตอบ: ' + data[i][2] + '\n\n';
  }
  
  // 3. สร้าง System Instruction
  var systemInstruction = 'คุณเป็น "น้องฝีมือ" ผู้ช่วย AI ของ สนพ.สุรินทร์\n' +
    'ตอบคำถามจากข้อมูลที่ให้เท่านั้น ถ้าไม่มีข้อมูลให้บอกว่า "กรุณาติดต่อ 044-511-259"\n' +
    'ตอบสุภาพ กระชับ ภาษาไทย\n\n' +
    '=== ข้อมูล FAQ ===\n' + faqContext;
  
  // 4. เรียก Gemini API
  var answer = callGeminiWithSystem(systemInstruction, userQuestion);
  
  // 5. บันทึก log
  logChat(userQuestion, answer);
  
  return answer;
}

/**
 * บันทึก Log การสนทนา

function logChat(question, answer) {

    logSheet.appendRow(['วันเวลา', 'คำถาม', 'คำตอบ']);
  }
  logSheet.appendRow([new Date(), question, answer]);
}

8.4 Workshop: ระบบสรุปเอกสารอัตโนมัติ

🏋️ Workshop (45 นาที)

สร้างระบบ: อัปโหลด PDF/Doc → Gemini สรุป → บันทึกผลลง Sheets

💻 โค้ด: ระบบสรุปเอกสารอัตโนมัติ
/**
 * ระบบสรุปเอกสารอัตโนมัติ
 * อ่านไฟล์จาก Google Drive → Gemini สรุป → บันทึกลง Sheets
 */

// ======= MAIN FUNCTION =======
function summarizeDocuments() {
  var folderId = 'YOUR_FOLDER_ID'; // โฟลเดอร์ที่เก็บเอกสาร
  var folder = DriveApp.getFolderById(folderId);
  var files = folder.getFiles();
  
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var resultSheet = ss.getSheetByName('สรุปเอกสาร');
  if (!resultSheet) {
    resultSheet = ss.insertSheet('สรุปเอกสาร');
    resultSheet.appendRow(['ลำดับ', 'ชื่อไฟล์', 'ประเภท', 'วันที่สรุป', 
                           'สรุปย่อ', 'ประเด็นสำคัญ', 'ข้อเสนอแนะ']);
  }
  
  var count = 0;
  while (files.hasNext()) {
    var file = files.next();
    var mimeType = file.getMimeType();
    
    // อ่านเนื้อหาไฟล์
    var content = extractFileContent(file, mimeType);
    if (!content) continue;
    
    // เรียก Gemini สรุป
    var summary = summarizeWithGemini(content, file.getName());
    
    // บันทึกลง Sheet
    count++;
    resultSheet.appendRow([
      count,
      file.getName(),
      mimeType,
      new Date(),
      summary.shortSummary,
      summary.keyPoints,
      summary.recommendations
    ]);
    
    // หน่วงเวลา (ป้องกัน rate limit)
    Utilities.sleep(2000);
  }
  
  SpreadsheetApp.getUi().alert('สรุปเอกสารเสร็จ ' + count + ' ไฟล์');
}

// ======= อ่านเนื้อหาไฟล์ =======
function extractFileContent(file, mimeType) {
  try {
    if (mimeType === 'application/pdf') {
      // แปลง PDF เป็น Google Doc แล้วอ่านข้อความ
      var blob = file.getBlob();
      var resource = { title: file.getName() + '_temp', mimeType: 'application/vnd.google-apps.document' };
      var tempDoc = Drive.Files.insert(resource, blob, { ocr: true });
      var doc = DocumentApp.openById(tempDoc.id);
      var text = doc.getBody().getText();
      Drive.Files.remove(tempDoc.id); // ลบไฟล์ชั่วคราว
      return text;
    } else if (mimeType === 'application/vnd.google-apps.document') {
      var doc = DocumentApp.openById(file.getId());
      return doc.getBody().getText();
    } else if (mimeType.indexOf('text/') === 0) {
      return file.getBlob().getDataAsString();
    }
    return null;
  } catch (e) {
    Logger.log('Error reading: ' + file.getName() + ' - ' + e);
    return null;
  }
}

// ======= เรียก Gemini สรุปเอกสาร =======
function summarizeWithGemini(content, fileName) {
  var systemInstruction = 
    'คุณเป็นผู้เชี่ยวชาญสรุปเอกสารราชการไทย\n' +
    'ตอบเป็น JSON format:\n' +
    '{\n' +
    '  "shortSummary": "สรุปสั้น 2-3 ประโยค",\n' +
    '  "keyPoints": "ประเด็นสำคัญ 3-5 ข้อ (คั่นด้วย |)",\n' +
    '  "recommendations": "ข้อเสนอแนะ (ถ้ามี)"\n' +
    '}';
  
  var userPrompt = 'สรุปเอกสาร "' + fileName + '":\n\n' + content.substring(0, 10000);
  
  var result = callGeminiWithSystem(systemInstruction, userPrompt);
  
  try {
    // ลบ markdown code block ถ้ามี
    result = result.replace(/```json\n?/g, '').replace(/```\n?/g, '');
    return JSON.parse(result);
  } catch (e) {
    return {
      shortSummary: result.substring(0, 200),
      keyPoints: '-',
      recommendations: '-'
    };
  }
}

✅ Best Practices — แนวปฏิบัติที่ดี
✅ ควรทำ
  • เก็บ API Key ใน Script Properties
  • ใช้ try-catch จัดการ error
  • หน่วงเวลา (sleep) ระหว่าง request
  • จำกัดความยาว input (ตัด text ที่ยาวเกิน)
  • ใช้ model ที่ถูกที่สุดที่เพียงพอ
  • บันทึก log ทุกการเรียก API
  • ตั้ง maxOutputTokens เท่าที่จำเป็น
❌ ไม่ควรทำ
  • ฝัง API Key ในโค้ดโดยตรง
  • เรียก API ใน loop โดยไม่หน่วงเวลา
  • ส่งข้อมูลส่วนบุคคล (PII) ไปยัง API
  • ใช้ Pro model กับงานง่ายๆ
  • ไม่จัดการ error (ปล่อยให้ crash)
  • ไม่ validate input จากผู้ใช้
  • เรียก API จากฝั่ง client (HTML)
🔧 การจัดการ Error ที่พบบ่อย
Error Code ความหมาย วิธีแก้
400 Bad Request (prompt ผิดรูปแบบ) ตรวจสอบ JSON payload
401 Unauthorized (API Key ผิด) ตรวจ API Key ใน Properties
429 Rate Limit Exceeded เพิ่ม sleep() หรือรอ 1 นาที
500 Server Error (Google) Retry อีก 2-3 ครั้ง
SAFETY Content ถูก block ปรับ prompt ให้เหมาะสม
โค้ด Retry Logic:
function callGeminiWithRetry(prompt, maxRetries) {
  maxRetries = maxRetries || 3;
  for (var i = 0; i < maxRetries; i++) {
    try {
      var result = callGemini(prompt);
      if (result.indexOf('Error') !== 0) return result;
    } catch (e) {
      Logger.log('Retry ' + (i + 1) + ': ' + e.toString());
    }
    Utilities.sleep(2000 * (i + 1)); // Exponential backoff
  }
  return 'ไม่สามารถเชื่อมต่อ Gemini ได้ กรุณาลองใหม่';
}
