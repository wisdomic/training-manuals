9.1 ระบบค้นหา + กรอง (Search & Filter)

Client-side Search — กรองข้อมูลในตาราง

วิธีนี้เหมาะกับข้อมูลน้อย (ไม่เกิน 500 แถว) — กรองโดยไม่ต้องเรียก Backend:

// กรองตาราง (Client-side)
function filterTable() {
  var input = document.getElementById('searchBox').value.toLowerCase();
  var rows = document.querySelectorAll('#dataTable tbody tr');
  
  rows.forEach(function(row) {
    var text = row.textContent.toLowerCase();
    row.style.display = text.includes(input) ? '' : 'none';
  });
}

// เรียกใช้เมื่อพิมพ์
document.getElementById('searchBox').addEventListener('keyup', filterTable);
Server-side Search — ค้นหาจาก Sheet

วิธีนี้เหมาะกับข้อมูลมาก — ค้นหาที่ Backend แล้วส่งผลลัพธ์กลับ:

// Code.gs — ค้นหาจาก Sheet
function searchTrainees(keyword) {
  try {
    var ss = SpreadsheetApp.openById(SHEET_ID);
    var sheet = ss.getSheetByName('TRAINEES');
    var data = sheet.getDataRange().getValues();
    var headers = data[0];
    var results = [];
    
    for (var i = 1; i < data.length; i++) {
      var rowText = data[i].join(' ').toLowerCase();
      if (rowText.indexOf(keyword.toLowerCase()) !== -1) {
        var obj = {};
        for (var j = 0; j < headers.length; j++) {
          obj[headers[j]] = data[i][j];
        }
        results.push(obj);
      }
    }
    return { success: true, data: results };
  } catch (err) {
    return { success: false, error: err.toString() };
  }
}

9.3 Line Notify แจ้งเตือน

ส่งข้อความแจ้งเตือนไปยัง Line OA เมื่อมีเหตุการณ์สำคัญ (เช่น มีคนลงทะเบียนใหม่):

Line Messaging API — Push Message
// ฟังก์ชันส่ง Line (ใช้ได้ทุกโปรเจกต์)
function _sendLine(message) {
  try {
    var token = 'YOUR_CHANNEL_ACCESS_TOKEN';
    var userId = 'YOUR_USER_ID';
    
    UrlFetchApp.fetch('https://api.line.me/v2/bot/message/push', {
      method: 'post',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': 'Bearer ' + token
      },
      payload: JSON.stringify({
        to: userId,
        messages: [{
          type: 'text',
          text: message
        }]
      }),
      muteHttpExceptions: true
    });
  } catch (e) {
    Logger.log('Line error: ' + e.toString());
  }
}
ตัวอย่างการใช้งาน
// แจ้งเตือนเมื่อมีคนลงทะเบียนใหม่

  var msg = '📋 ลงทะเบียนใหม่!\n';
  msg += '👤 ชื่อ: ' + name + '\n';
  msg += '📚 หลักสูตร: ' + courseName + '\n';

ขั้นตอนตั้งค่า Line Messaging API
  1. ไปที่ developers.line.biz
  2. สร้าง Provider > สร้าง Channel (Messaging API)
  3. คัดลอก Channel Access Token (long-lived)
  4. ใน Basic Settings — คัดลอก Your User ID
  5. ใส่ Token + User ID ในฟังก์ชัน _sendLine()
  6. ใน GAS: เพิ่ม scope "external_request" ใน appsscript.json
💡 เคล็ดลับ: เก็บ Token ใน Sheet SETTINGS แทนการ hardcode — ง่ายต่อการเปลี่ยนภายหลัง

9.5 ระบบ Login + สิทธิ์ผู้ใช้

Simple Login ด้วย PropertiesService
// Code.gs — ระบบ Login
function login(username, password) {
  try {
    var ss = SpreadsheetApp.openById(SHEET_ID);
    var sheet = ss.getSheetByName('USERS');
    var data = sheet.getDataRange().getValues();
    
    for (var i = 1; i < data.length; i++) {
      if (data[i][1] === username) {
        // ตรวจรหัสผ่าน (hash)
        var hash = Utilities.computeDigest(
          Utilities.DigestAlgorithm.SHA_256,
          password + data[i][4]  // password + salt
        );
        var hashStr = hash.map(function(b) {
          return ('0' + (b & 0xFF).toString(16)).slice(-2);
        }).join('');
        
        if (hashStr === data[i][2]) {
          return {
            success: true,
            data: {
              user_id: data[i][0],
              username: data[i][1],
              role: data[i][3]  // admin / user
            }
          };
        }
      }
    }
    return { success: false, error: 'ชื่อผู้ใช้หรือรหัสผ่านไม่ถูกต้อง' };
  } catch (err) {
    return { success: false, error: err.toString() };
  }
}

// Frontend — เก็บ session ใน localStorage
function handleLogin() {
  var username = document.getElementById('username').value;
  var password = document.getElementById('password').value;
  
  google.script.run
    .withSuccessHandler(function(result) {
      if (result.success) {
        // เก็บข้อมูล login
        localStorage.setItem('user', JSON.stringify(result.data));
        localStorage.setItem('loginTime', new Date().getTime());
        showAdminPanel();
      } else {
        alert(result.error);
      }
    })
    .login(username, password);
}

// ตรวจสอบ login เมื่อเปิดหน้า
function checkLogin() {
  var user = localStorage.getItem('user');
  if (!user) {
    showLoginForm();
    return;
  }
  // ตรวจ session timeout (8 ชม.)
  var loginTime = localStorage.getItem('loginTime');
  if (new Date().getTime() - loginTime > 8 * 60 * 60 * 1000) {
    localStorage.clear();
    showLoginForm();
    return;
  }
  showAdminPanel();
}

// Logout
function logout() {
  localStorage.clear();
  showLoginForm();
}

<!-- JavaScript Validation -->
<script>
function validateForm() {
  var phone = document.getElementById('phone').value;
  var idCard = document.getElementById('idCard').value;
  
  // ตรวจเบอร์โทร
  if (!/^0[0-9]{9}$/.test(phone)) {
    alert('เบอร์โทรไม่ถูกต้อง (ต้องขึ้นต้นด้วย 0 และมี 10 หลัก)');
    return false;
  }
  
  // ตรวจเลขบัตร (Luhn Algorithm)
  if (!validateIdCard(idCard)) {
    alert('เลขบัตรประชาชนไม่ถูกต้อง');
    return false;
  }
  
  return true;
}

function validateIdCard(id) {
  if (id.length !== 13) return false;
  var sum = 0;
  for (var i = 0; i < 12; i++) {
    sum += parseInt(id[i]) * (13 - i);
  }
  var check = (11 - (sum % 11)) % 10;
  return check === parseInt(id[12]);
}
</script>
Server-side Validation (GAS)
// Code.gs — ตรวจสอบข้อมูลที่ Backend
function validateInput(data) {
  var errors = [];
  
  // ตรวจค่าว่าง
  if (!data.first_name || !data.first_name.trim()) {
    errors.push('กรุณากรอกชื่อ');
  }
  if (!data.last_name || !data.last_name.trim()) {
    errors.push('กรุณากรอกนามสกุล');
  }
  
  // ตรวจความยาว
  if (data.first_name && data.first_name.length > 100) {
    errors.push('ชื่อยาวเกินไป');
  }
  
  // ตรวจ format
  if (data.phone && !/^0[0-9]{9}$/.test(data.phone)) {
    errors.push('เบอร์โทรไม่ถูกต้อง');
  }
  
  // ตรวจ XSS
  var dangerousPattern = /[<>"'&]/;
  if (dangerousPattern.test(data.first_name) || dangerousPattern.test(data.last_name)) {
    errors.push('ข้อมูลมีอักขระไม่อนุญาต');
  }
  
  return errors;
}
XSS Prevention
// ฟังก์ชัน Escape HTML (ป้องกัน XSS)
function escapeHtml(text) {
  if (!text) return '';
  return String(text)
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#39;');
}

// Code.gs — จำกัด 5 ครั้ง/5 นาที
function checkRateLimit(action, limit, windowMinutes) {
  var cache = CacheService.getScriptCache();
  var key = 'rate_' + action + '_' + Session.getActiveUser().getEmail();
  var count = parseInt(cache.get(key) || '0');
  
  if (count >= limit) {
    return false;  // เกินจำนวนที่กำหนด
  }
  
  cache.put(key, String(count + 1), windowMinutes * 60);
  return true;  // อนุญาต
}
