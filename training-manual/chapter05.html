พื้นฐาน Google Apps Script

ระยะเวลา: 2 ชั่วโมง | วันที่ 2

🎯 วัตถุประสงค์การเรียนรู้
  • เข้าใจว่า Google Apps Script คืออะไรและทำงานอย่างไร
  • สร้าง Web App ด้วย doGet() / doPost() ได้
  • อ่าน/เขียนข้อมูล Google Sheets ด้วย SpreadsheetApp ได้

5.1 Google Apps Script คืออะไร

Google Apps Script (GAS) คือ JavaScript runtime ของ Google ที่ทำงานบน Cloud โดยไม่ต้องติดตั้ง Server เอง

☁️ ทำงานบน Cloud

ไม่ต้องเช่า Server, ไม่ต้องติดตั้งอะไร ใช้ได้ทันทีผ่าน Browser

🔗 เชื่อมกับ Google Services

Sheets, Drive, Gmail, Calendar, Forms ใช้ได้หมดโดยไม่ต้อง config

📝 เป็น JavaScript

ใช้ภาษา JavaScript (ES6+) เขียนได้เลย ไม่ต้องเรียนภาษาใหม่

⚡ สรุป: GAS = JavaScript + Google Services + Free Hosting + Free Database (Sheets) = สร้าง Web App ได้ฟรี!

5.2 Script Editor + โครงสร้างโปรเจกต์

เข้า Script Editor: script.google.com หรือจาก Google Sheets > Extensions > Apps Script

โครงสร้างไฟล์ในโปรเจกต์ GAS
ไฟล์หน้าที่ตัวอย่าง
.gsBackend (Server-side)Code.gs, Database.gs, Utils.gs
.htmlFrontend (Client-side)index.html, admin.html, style.html
appsscript.jsonConfig (ตั้งค่าโปรเจกต์)scopes, timezone, webapp
ตัวอย่าง appsscript.json
{
  "timeZone": "Asia/Bangkok",
  "dependencies": {},
  "exceptionLogging": "STACKDRIVER",
  "runtimeVersion": "V8",
  "webapp": {
    "executeAs": "USER_DEPLOYING",
    "access": "ANYONE"
  },
  "oauthScopes": [
    "https://www.googleapis.com/auth/spreadsheets",
    "https://www.googleapis.com/auth/script.external_request"
  ]
}
💡 เคล็ดลับ: แยกไฟล์ .gs ตามหน้าที่ เช่น Code.gs (doGet/doPost), Database.gs (อ่าน/เขียน), Utils.gs (ฟังก์ชันช่วย) — ทำให้โค้ดจัดการง่าย!

5.3 doGet() / doPost() — Web App

GAS ใช้ 2 ฟังก์ชันหลักในการรับ request จากผู้ใช้:

doGet(e)
  • ทำงานเมื่อผู้ใช้เปิดเว็บ
  • รับข้อมูลผ่าน URL parameters
  • ใช้แสดงหน้า HTML
doPost(e)
  • ทำงานเมื่อผู้ใช้ส่งข้อมูล
  • รับข้อมูลผ่าน request body
  • ใช้บันทึกข้อมูล / API
ตัวอย่าง doGet() — แสดงหน้าเว็บ
// Code.gs
function doGet(e) {
  // รับ parameter จาก URL เช่น ?page=admin
  var page = e.parameter.page || 'index';
  
  // แสดงหน้า HTML ตาม parameter
  return HtmlService.createHtmlOutputFromFile(page)
    .setTitle('ระบบลงทะเบียนฝึกอบรม')
    .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
}
ตัวอย่าง doPost() — รับข้อมูล API
// Code.gs
function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var action = data.action;
    
    if (action === 'register') {
      var result = registerTrainee(data);
      return ContentService.createTextOutput(
        JSON.stringify({ success: true, data: result })
      ).setMimeType(ContentService.MimeType.JSON);
    }
  } catch (err) {
    return ContentService.createTextOutput(
      JSON.stringify({ success: false, error: err.toString() })
    ).setMimeType(ContentService.MimeType.JSON);
  }
}

5.4 SpreadsheetApp — อ่าน/เขียนข้อมูล

SpreadsheetApp คือ API หลักสำหรับอ่าน/เขียนข้อมูลใน Google Sheets:

เปิด Spreadsheet + อ่านข้อมูล
// เปิด Spreadsheet ด้วย ID
var ss = SpreadsheetApp.openById('SPREADSHEET_ID_HERE');

// เลือก Sheet ตามชื่อ
var sheet = ss.getSheetByName('TRAINEES');

// อ่านข้อมูลทั้งหมด
var data = sheet.getDataRange().getValues();
// data = [[หัวคอลัมน์], [แถว1], [แถว2], ...]

// อ่านเฉพาะหัวคอลัมน์
var headers = data[0]; // ['trainee_id', 'prefix', 'first_name', ...]

// อ่านข้อมูล (ไม่รวมหัว)
var rows = data.slice(1);
เขียนข้อมูล — เพิ่มแถวใหม่
// เพิ่มแถวใหม่ท้ายตาราง
sheet.appendRow([
  'T002',           // trainee_id
  'นางสาว',         // prefix
  'สุดา',           // first_name
  'รักเรียน',       // last_name
  '3201xxxxx',     // id_card
  '09x-xxx-xxxx', // phone
  'suda@mail.com', // email
  new Date()       // created_at
]);
เขียนข้อมูล — อัปเดตหลายเซลล์
// อัปเดตข้อมูลในช่วง A2:C4
var values = [
  ['T001', 'นาย', 'สมชาย'],
  ['T002', 'นางสาว', 'สุดา'],
  ['T003', 'นาง', 'มาลี']
];
sheet.getRange(2, 1, 3, 3).setValues(values);
// getRange(แถวเริ่ม, คอลัมน์เริ่ม, จำนวนแถว, จำนวนคอลัมน์)
📌 ฟังก์ชันที่ใช้บ่อย: openById(), getSheetByName(), getDataRange(), getValues(), appendRow(), getRange().setValues()

5.5 HtmlService — แสดงผล HTML

HtmlService ใช้สร้างหน้าเว็บจากไฟล์ .html ในโปรเจกต์:

วิธีที่ 1: createHtmlOutputFromFile (ง่ายสุด)
function doGet() {
  return HtmlService.createHtmlOutputFromFile('index')
    .setTitle('ระบบลงทะเบียน')
    .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
}
วิธีที่ 2: createTemplateFromFile (ใส่ข้อมูลได้)
function doGet() {
  var template = HtmlService.createTemplateFromFile('index');
  template.userName = 'สมชาย';  // ส่งตัวแปรไปหน้า HTML
  template.courses = getCourses();
  return template.evaluate()
    .setTitle('ระบบลงทะเบียน');
}

// ในไฟล์ index.html ใช้:
// <p>สวัสดี <?= userName ?></p>
ฟังก์ชัน include() — แยกไฟล์ CSS/JS
// Code.gs — ฟังก์ชันช่วย include
function include(filename) {
  return HtmlService.createHtmlOutputFromFile(filename).getContent();
}

// index.html — เรียกใช้
// <head>
//   <?!= include('style') ?>    <!-- รวมไฟล์ style.html -->
// </head>
// <body>
//   ...
//   <?!= include('script') ?>   <!-- รวมไฟล์ script.html -->
// </body>
💡 Best Practice: แยก CSS ไว้ในไฟล์ style.html, JavaScript ไว้ใน script.html แล้วใช้ include() รวมเข้ามา — โค้ดจัดการง่ายขึ้นมาก!

5.6 google.script.run — เชื่อม Frontend กับ Backend

google.script.run คือวิธีเรียกฟังก์ชัน Backend (.gs) จาก Frontend (.html):

โครงสร้างการเรียก
google.script.run
  .withSuccessHandler(onSuccess)   // เมื่อสำเร็จ
  .withFailureHandler(onError)     // เมื่อเกิด error
  .functionName(arg1, arg2);       // ชื่อฟังก์ชันใน .gs
ตัวอย่างจริง: โหลดรายชื่อหลักสูตร
// ===== Backend: Code.gs =====
function getCourses() {
  var ss = SpreadsheetApp.openById('SHEET_ID');
  var sheet = ss.getSheetByName('COURSES');
  var data = sheet.getDataRange().getValues();
  var headers = data[0];
  var courses = [];
  
  for (var i = 1; i < data.length; i++) {
    var obj = {};
    for (var j = 0; j < headers.length; j++) {
      obj[headers[j]] = data[i][j];
    }
    courses.push(obj);
  }
  return courses;
}

// ===== Frontend: index.html =====
<script>
// โหลดหลักสูตรเมื่อเปิดหน้า
document.addEventListener('DOMContentLoaded', function() {
  google.script.run
    .withSuccessHandler(function(courses) {
      // สำเร็จ — แสดงข้อมูล
      var html = '';
      courses.forEach(function(c) {
        html += '<option value="' + c.course_id + '">' 
              + c.course_name + '</option>';
      });
      document.getElementById('courseSelect').innerHTML = html;
    })
    .withFailureHandler(function(err) {
      // เกิด error
      alert('เกิดข้อผิดพลาด: ' + err.message);
    })
    .getCourses();
});
</script>
⚠️ ข้อควรระวัง:
  • google.script.run เป็น Asynchronous — ต้องรอ callback ไม่สามารถ return ค่าได้ทันที
  • ส่งได้เฉพาะ JSON-serializable data (ห้ามส่ง Date object, function)
  • ขนาดข้อมูลที่ส่งได้ไม่เกิน 256KB

5.7 Deploy Web App ครั้งแรก

ขั้นตอนการ Deploy Web App ให้คนอื่นเข้าถึงได้:

📋 ขั้นตอน Deploy
  1. Deploy > New deployment — คลิกปุ่ม Deploy ด้านบนขวา
  2. Select type > Web app — เลือกประเภท Web app
  3. Description — ใส่คำอธิบาย เช่น "v1 - เปิดใช้งานครั้งแรก"
  4. Execute as: Me — ให้ script ทำงานในนามของเรา (เข้าถึง Sheets ได้)
  5. Who has access: Anyone — ให้ทุกคนเข้าถึงได้ (ไม่ต้อง login Google)
  6. Deploy — คลิก Deploy แล้วจะได้ URL
URL ที่ได้หลัง Deploy
https://script.google.com/macros/s/AKfycbx.../exec
🔄 อัปเดต Deploy

แก้โค้ดแล้วต้อง Deploy ใหม่:

Deploy > Manage deployments > Edit > New version > Deploy

⚠️ ข้อควรระวัง
  • แก้โค้ดแล้วต้อง Deploy ใหม่ทุกครั้ง
  • ถ้าไม่ Deploy ใหม่ จะยังใช้โค้ดเวอร์ชันเก่า
  • URL จะเปลี่ยนทุกครั้งที่ New deployment
🎉 สำเร็จ! ตอนนี้คุณมี Web App ที่ทุกคนเข้าถึงได้ผ่าน URL — ฟรี ไม่ต้องเช่า Server!
