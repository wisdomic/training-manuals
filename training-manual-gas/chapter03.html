บทที่ 3: ฟังก์ชัน CRUD

สร้าง/อ่าน/แก้ไข/ลบ ข้อมูลใน Google Sheets อย่างเป็นระบบ

🎯 วัตถุประสงค์การเรียนรู้

  • เข้าใจแนวคิด CRUD (Create, Read, Update, Delete)
  • เขียนฟังก์ชัน appendRow() เพิ่มข้อมูลใหม่ได้
  • เขียนฟังก์ชันค้นหาข้อมูล (findRow) ได้
  • แก้ไขและลบข้อมูลในแถวที่ต้องการได้

📚 CRUD คืออะไร?

CRUD คือ 4 การทำงานพื้นฐานกับข้อมูลทุกระบบ:

🔵 R

🟡 U

Update

🔴 D

Delete
ลบข้อมูล

🟢 Create — เพิ่มข้อมูลใหม่

function createRecord(name, province, phone) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("สมาชิก");
  
  // สร้าง ID อัตโนมัติ
  var lastRow = sheet.getLastRow();
  var newId = lastRow;  // ID = ลำดับแถว
  
  // เพิ่มข้อมูลแถวใหม่
  var timestamp = new Date();
  sheet.appendRow([newId, name, province, phone, timestamp]);
  
  Logger.log("เพิ่มสมาชิกสำเร็จ: " + name + " (ID: " + newId + ")");
  return newId;
}

// ตัวอย่างการใช้งาน
function testCreate() {
  createRecord("นายทดสอบ ทดลอง", "สุรินทร์", "081-111-2222");
  createRecord("นางสาวใจดี มีสุข", "ปราจีนบุรี", "089-333-4444");
}

🔵 Read — อ่าน/ค้นหาข้อมูล

// ค้นหาแถวตาม ID
function findRowById(id) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("สมาชิก");
  var data = sheet.getDataRange().getValues();
  
  for (var i = 1; i < data.length; i++) {
    if (String(data[i][0]) === String(id)) {
      return {
        rowIndex: i + 1,  // +1 เพราะ Sheet เริ่มนับแถวที่ 1
        data: data[i]
      };
    }
  }
  return null;  // ไม่พบ
}

// ค้นหาตามชื่อ (แบบ contains)
function findByName(keyword) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("สมาชิก");
  var data = sheet.getDataRange().getValues();
  var results = [];
  
  for (var i = 1; i < data.length; i++) {
    if (String(data[i][1]).indexOf(keyword) !== -1) {
      results.push({
        rowIndex: i + 1,
        data: data[i]
      });
    }
  }
  return results;
}

// อ่านข้อมูลทั้งหมด
function readAllRecords() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("สมาชิก");
  var data = sheet.getDataRange().getValues();
  var headers = data[0];
  var records = [];
  
  for (var i = 1; i < data.length; i++) {
    var record = {};
    for (var j = 0; j < headers.length; j++) {
      record[headers[j]] = data[i][j];
    }
    records.push(record);
  }
  return records;
}

🟡 Update — แก้ไขข้อมูล

function updateRecord(id, newName, newProvince, newPhone) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("สมาชิก");
  var found = findRowById(id);
  
  if (!found) {
    Logger.log("ไม่พบข้อมูล ID: " + id);
    return false;
  }
  
  var rowNum = found.rowIndex;
  
  // อัปเดตเฉพาะฟิลด์ที่ส่งมา
  if (newName) sheet.getRange(rowNum, 2).setValue(newName);
  if (newProvince) sheet.getRange(rowNum, 3).setValue(newProvince);
  if (newPhone) sheet.getRange(rowNum, 4).setValue(newPhone);
  
  Logger.log("อัปเดตสำเร็จ ID: " + id);
  return true;
}

// ตัวอย่าง
function testUpdate() {
  updateRecord(1, "นายปัญญา ผาธรรม (อ.ปู)", "สุรินทร์", "088-377-8987");
}

🔴 Delete — ลบข้อมูล

function deleteRecord(id) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("สมาชิก");
  var found = findRowById(id);
  
  if (!found) {
    Logger.log("ไม่พบข้อมูล ID: " + id);
    return false;
  }
  
  // ลบแถว
  sheet.deleteRow(found.rowIndex);
  Logger.log("ลบสำเร็จ ID: " + id + " (ชื่อ: " + found.data[1] + ")");
  return true;
}

// ⚠️ ระวัง: ลบหลายแถวต้องลบจากล่างขึ้นบน
function deleteMultipleRows(ids) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("สมาชิก");
  var data = sheet.getDataRange().getValues();
  
  // หาแถวที่ต้องลบ แล้วเรียงจากมากไปน้อย
  var rowsToDelete = [];
  for (var i = 1; i < data.length; i++) {
    if (ids.indexOf(data[i][0]) !== -1) {
      rowsToDelete.push(i + 1);
    }
  }
  rowsToDelete.sort(function(a, b) { return b - a; });
  
  // ลบจากล่างขึ้นบน (เพื่อไม่ให้ index เลื่อน)
  for (var i = 0; i < rowsToDelete.length; i++) {
    sheet.deleteRow(rowsToDelete[i]);
  }
  Logger.log("ลบ " + rowsToDelete.length + " รายการสำเร็จ");
}
⚠️ ข้อควรระวังเรื่องการลบ
  • ลบแล้วกู้คืนไม่ได้ (ยกเว้นใช้ Version History ของ Sheets)
  • ลบหลายแถว ต้องลบจากแถวล่างขึ้นบน ไม่งั้น index จะเลื่อน
  • ควรมีระบบยืนยันก่อนลบ (confirm)

🔧 รวม CRUD เป็นระบบเดียว

// ===== ระบบจัดการสมาชิก CRUD =====

var SHEET_NAME = "สมาชิก";

function getSheet() {
  return SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
}

function generateId() {
  var sheet = getSheet();
  var lastRow = sheet.getLastRow();
  if (lastRow <= 1) return 1;
  var lastId = sheet.getRange(lastRow, 1).getValue();
  return Number(lastId) + 1;
}

function crud_create(data) {
  var sheet = getSheet();
  var id = generateId();
  sheet.appendRow([id, data.name, data.province, data.phone, new Date()]);
  return { success: true, id: id, message: "เพิ่มสำเร็จ" };
}

function crud_read(id) {
  var found = findRowById(id);
  if (!found) return { success: false, message: "ไม่พบข้อมูล" };
  return { success: true, data: found.data };
}

function crud_update(id, data) {
  var sheet = getSheet();
  var found = findRowById(id);
  if (!found) return { success: false, message: "ไม่พบข้อมูล" };
  
  var row = found.rowIndex;
  if (data.name) sheet.getRange(row, 2).setValue(data.name);
  if (data.province) sheet.getRange(row, 3).setValue(data.province);
  if (data.phone) sheet.getRange(row, 4).setValue(data.phone);
  return { success: true, message: "อัปเดตสำเร็จ" };
}

function crud_delete(id) {
  var sheet = getSheet();
  var found = findRowById(id);
  if (!found) return { success: false, message: "ไม่พบข้อมูล" };
  
  sheet.deleteRow(found.rowIndex);
  return { success: true, message: "ลบสำเร็จ" };
}

function lab5_testCRUD() {
  // Create — เพิ่ม 3 คน
  var r1 = crud_create({name: "สมชาย ทดสอบ", province: "สุรินทร์", phone: "081-111-1111"});
  var r2 = crud_create({name: "สมหญิง ทดลอง", province: "ยโสธร", phone: "082-222-2222"});
  var r3 = crud_create({name: "สมศักดิ์ ใจดี", province: "ปราจีนบุรี", phone: "083-333-3333"});
  Logger.log("สร้าง: " + r1.message + " ID=" + r1.id);
  Logger.log("สร้าง: " + r2.message + " ID=" + r2.id);
  Logger.log("สร้าง: " + r3.message + " ID=" + r3.id);
  
  // Read — อ่านข้อมูล ID 1
  var read = crud_read(1);
  Logger.log("อ่าน ID 1: " + JSON.stringify(read));
  
  // Update — แก้ไข ID 2
  var upd = crud_update(2, {name: "สมหญิง แก้ไขแล้ว", phone: "099-999-9999"});
  Logger.log("แก้ไข: " + upd.message);
  
  // Delete — ลบ ID 3
  var del = crud_delete(3);
  Logger.log("ลบ: " + del.message);
  
  Logger.log("\n===== ทดสอบ CRUD เสร็จสิ้น =====");

💡 Tips

  • ID ควรเป็นตัวเลขไม่ซ้ำ — ใช้ lastRow หรือ UUID
  • ค้นหาแบบ String — แปลงเป็น String() ก่อนเปรียบเทียบเสมอ (เพราะ Sheets อาจเก็บเป็นตัวเลข)
  • ประสิทธิภาพ: อ่านข้อมูลทั้งหมดมาเก็บใน Array แล้วค้นหาใน Array (เร็วกว่าอ่านทีละเซลล์มาก)
  • Return Object: ให้ฟังก์ชันคืนค่า {success, message, data} เพื่อให้เรียกใช้งานง่าย
