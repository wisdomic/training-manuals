บทที่ 10: Workshop — สร้างระบบลงทะเบียนกิจกรรมครบวงจร

รวมทุกอย่างที่เรียนมา สร้างระบบจริงที่ใช้งานได้ (Form → Sheet → Email → Web App)

🎯 วัตถุประสงค์การเรียนรู้

  • สร้างระบบลงทะเบียนกิจกรรมที่ใช้งานได้จริง
  • เชื่อมต่อ Google Form → Sheet → Email → Web App
  • ประยุกต์ใช้ CRUD, Trigger, MailApp, HTML Service ร่วมกัน
  • Deploy ระบบให้คนอื่นใช้งานได้

🏗️ ภาพรวมระบบที่จะสร้าง

📋 ระบบลงทะเบียนกิจกรรม ประกอบด้วย:

  1. Google Form — ฟอร์มลงทะเบียน (ชื่อ, หน่วยงาน, อีเมล, โทร)
  2. Google Sheet — เก็บข้อมูลผู้ลงทะเบียนทั้งหมด
  3. Email อัตโนมัติ — ส่งยืนยันเมื่อลงทะเบียนสำเร็จ
  4. Web App (Admin) — หน้าดูรายชื่อ + จัดการข้อมูล
  5. Trigger — ส่งรายงานสรุปทุกเช้า
Flow: ผู้ใช้กรอก Form → ข้อมูลลง Sheet อัตโนมัติ → GAS ส่ง Email ยืนยัน → Admin ดูข้อมูลผ่าน Web App → รายงานสรุปทุกเช้า

var SHEET_NAME = "registrations";
var ADMIN_EMAIL = "wisdomssk@gmail.com";

// ===== แสดงหน้าเว็บ Admin =====
function doGet(e) {
  var page = e.parameter.page || "admin";
  
  if (page === "admin") {
    return HtmlService.createHtmlOutputFromFile("admin")
      .setTitle("Admin - ระบบลงทะเบียน")
      .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
  }
  
  // API mode
  if (page === "api") {
    return handleApi(e);
  }
  
  return HtmlService.createHtmlOutput("404 - ไม่พบหน้าที่ต้องการ");
}

function handleApi(e) {
  var action = e.parameter.action || "count";
  var result = {};
  
  if (action === "count") {
    var sheet = getSheet();
    result = { count: sheet.getLastRow() - 1 };
  } else if (action === "list") {
    result = { data: getRegistrations() };
  }
  
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

// ===== HELPER FUNCTIONS =====
function getSheet() {
  return SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
}

// ===== CRUD =====
function getRegistrations() {
  var sheet = getSheet();
  var data = sheet.getDataRange().getValues();
  var registrations = [];
  
  for (var i = 1; i < data.length; i++) {
    registrations.push({
      row: i + 1,
      timestamp: data[i][0],
      name: data[i][1],
      organization: data[i][2],
      email: data[i][3],
      phone: data[i][4],
      status: data[i][5] || "รอยืนยัน",
      emailSent: data[i][6] || ""
    });
  }
  return registrations;
}

function updateStatus(row, status) {
  var sheet = getSheet();
  sheet.getRange(row, 6).setValue(status);
  Logger.log("อัปเดตสถานะแถว " + row + " เป็น " + status);
  return { success: true, message: "อัปเดตสถานะสำเร็จ" };
}

function deleteRegistration(row) {
  var sheet = getSheet();
  var name = sheet.getRange(row, 2).getValue();
  sheet.deleteRow(row);
  Logger.log("ลบข้อมูล: " + name);
  return { success: true, message: "ลบ " + name + " สำเร็จ" };
}

function getStats() {
  var registrations = getRegistrations();
  var total = registrations.length;
  var confirmed = 0;
  var pending = 0;
  var cancelled = 0;
  
  for (var i = 0; i < registrations.length; i++) {
    var status = registrations[i].status;
    if (status === "ยืนยันแล้ว") confirmed++;
    else if (status === "ยกเลิก") cancelled++;
    else pending++;
  }
  
  return {
    total: total,
    confirmed: confirmed,
    pending: pending,
    cancelled: cancelled
  };
}

⏰ ขั้นตอนที่ 6: ตั้ง Trigger

// ===== TRIGGERS =====

function setupTriggers() {
  // ลบ Trigger เดิม
  var triggers = ScriptApp.getProjectTriggers();
  for (var i = 0; i < triggers.length; i++) {
    ScriptApp.deleteTrigger(triggers[i]);
  }
  
  // 1. onFormSubmit — ส่ง Email เมื่อมีคนลงทะเบียน
  ScriptApp.newTrigger("onFormSubmit")
    .forSpreadsheet(SpreadsheetApp.getActiveSpreadsheet())
    .onFormSubmit()
    .create();
  
  // 2. Daily Report — ส่งรายงานทุกเช้า 08:00
  ScriptApp.newTrigger("sendDailyReport")
    .timeBased()
    .atHour(8)
    .everyDays(1)
    .create();
  
  Logger.log("ตั้ง Trigger สำเร็จ! (2 ตัว)");
}

function sendDailyReport() {
  var stats = getStats();
  var message = "📊 รายงานประจำวัน - ระบบลงทะเบียน\n\n";
  message += "📅 " + new Date().toLocaleDateString("th-TH") + "\n\n";
  message += "👥 ผู้ลงทะเบียนทั้งหมด: " + stats.total + " คน\n";
  message += "✅ ยืนยันแล้ว: " + stats.confirmed + " คน\n";
  message += "⏳ รอยืนยัน: " + stats.pending + " คน\n";
  message += "❌ ยกเลิก: " + stats.cancelled + " คน\n";
  
  MailApp.sendEmail(ADMIN_EMAIL, "📊 รายงานลงทะเบียน (" + stats.total + " คน)", message);
  Logger.log("ส่งรายงานประจำวันสำเร็จ");
}

// 2. ส่งแจ้งเตือน LINE
function sendLineNotify(message) {
  var token = "YOUR_LINE_CHANNEL_TOKEN";
  var userId = "YOUR_LINE_USER_ID";
  
  UrlFetchApp.fetch("https://api.line.me/v2/bot/message/push", {
    method: "post",
    headers: {
      "Content-Type": "application/json",
      "Authorization": "Bearer " + token
    },
    payload: JSON.stringify({
      to: userId,
      messages: [{ type: "text", text: message }]
    })
  });
}

// 3. ตรวจ Duplicate (ซ้ำ)
function checkDuplicate(email) {
  var registrations = getRegistrations();
  for (var i = 0; i < registrations.length; i++) {
    if (registrations[i].email === email) return true;
  }
  return false;
}

function generateQRUrl(data) {
  var text = "ชื่อ: " + data.name + " | หน่วยงาน: " + data.org;
  return "https://api.qrserver.com/v1/create-qr-code/?size=200x200&data=" + encodeURIComponent(text);
}

💡 Tips สำหรับ Workshop

  • ทำทีละขั้น: อย่าเขียนทั้งหมดแล้วค่อยทดสอบ — ทำทีละส่วนแล้วทดสอบไปเรื่อยๆ
  • ดู Execution Log: ถ้ามี Error ให้ดูที่ Executions (เมนูด้านซ้าย)
  • ใช้ /dev URL ทดสอบ: ไม่ต้อง deploy ใหม่ทุกรอบ
  • เอา Logger.log ช่วย Debug: ใส่ทุกจุดที่สงสัย
  • ขอความช่วยเหลือ: ถ้าติดปัญหา ยกมือถามวิทยากรได้เลย!
