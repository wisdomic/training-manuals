📖 Google Apps Script บทที่ 5: Trigger

บทที่ 5: Trigger — ตั้งเวลาทำงานอัตโนมัติ

ให้โปรแกรมทำงานเองโดยไม่ต้องกดปุ่ม — ทั้งตามเวลาและตามเหตุการณ์

🎯 วัตถุประสงค์การเรียนรู้

  • เข้าใจแนวคิด Trigger (ตัวจุดชนวน) ใน GAS
  • สร้าง Time-based Trigger (ทำงานตามเวลา)
  • สร้าง Event-based Trigger (onOpen, onEdit, onFormSubmit)
  • จัดการ Trigger ด้วยโค้ด (สร้าง/ลบ)

📚 Trigger คืออะไร?

Trigger คือ "นาฬิกาปลุก" ของ GAS — สั่งให้ฟังก์ชันทำงาน โดยอัตโนมัติ ตามเงื่อนไขที่กำหนด เช่น:

  • ทุกวันเวลา 08:00 → ส่งรายงานอีเมล
  • ทุก 1 ชั่วโมง → ตรวจสอบข้อมูลใหม่
  • เมื่อเปิด Sheet → แสดงเมนูพิเศษ
  • เมื่อแก้ไขเซลล์ → คำนวณใหม่อัตโนมัติ

📋 ประเภทของ Trigger

ประเภท ทำงานเมื่อ ตัวอย่างการใช้
Time-based ตามเวลาที่กำหนด ส่งรายงานทุกเช้า, สำรองข้อมูลทุกคืน
onOpen เปิด Spreadsheet สร้างเมนูพิเศษ, แสดงข้อความต้อนรับ
onEdit แก้ไขเซลล์ ใส่ Timestamp อัตโนมัติ, คำนวณราคา
onFormSubmit ส่ง Google Form ส่งอีเมลยืนยัน, สร้างเอกสาร
onChange โครงสร้าง Sheet เปลี่ยน เพิ่ม/ลบ Sheet, ย้ายแถว

⏰ Time-based Trigger (Cron)

// สร้าง Trigger ทำงานทุกวัน เวลา 08:00-09:00
function createDailyTrigger() {
  ScriptApp.newTrigger("sendDailyReport")
    .timeBased()
    .atHour(8)           // 08:00-09:00 น.
    .everyDays(1)        // ทุก 1 วัน
    .create();
  Logger.log("สร้าง Daily Trigger สำเร็จ!");
}

// ทำงานทุก 1 ชั่วโมง
function createHourlyTrigger() {
  ScriptApp.newTrigger("checkNewData")
    .timeBased()
    .everyHours(1)
    .create();
}

// ทำงานทุก 30 นาที
function createMinuteTrigger() {
  ScriptApp.newTrigger("quickCheck")
    .timeBased()
    .everyMinutes(30)    // 1, 5, 10, 15, 30
    .create();
}

// ทำงานทุกสัปดาห์ วันจันทร์
function createWeeklyTrigger() {
  ScriptApp.newTrigger("sendWeeklyReport")
    .timeBased()
    .onWeekDay(ScriptApp.WeekDay.MONDAY)
    .atHour(9)
    .create();
}

📋 Event-based Trigger

onOpen — เมื่อเปิด Spreadsheet

// Simple Trigger (ไม่ต้องสร้าง — แค่ตั้งชื่อ onOpen)
function onOpen() {
  var ui = SpreadsheetApp.getUi();
  
  // สร้างเมนูพิเศษ
  ui.createMenu("🛠️ เครื่องมือ")
    .addItem("ส่งรายงาน", "sendReport")
    .addItem("สำรองข้อมูล", "backupData")
    .addSeparator()
    .addItem("เกี่ยวกับ", "showAbout")
    .addToUi();
    
  Logger.log("สร้างเมนูสำเร็จ");
}

function showAbout() {
  SpreadsheetApp.getUi().alert(
    "ระบบจัดการข้อมูล v1.0\nพัฒนาโดย: นายปัญญา ผาธรรม\nสนพ.สุรินทร์"
  );
}

onEdit — เมื่อแก้ไขเซลล์

// Simple Trigger — ใส่ Timestamp อัตโนมัติ
function onEdit(e) {
  var sheet = e.source.getActiveSheet();
  var range = e.range;
  
  // ถ้าแก้ไขคอลัมน์ B (ชื่อ) → ใส่เวลาในคอลัมน์ F
  if (range.getColumn() === 2 && sheet.getName() === "สมาชิก") {
    var row = range.getRow();
    if (row > 1) {  // ไม่ใช่ header
      sheet.getRange(row, 6).setValue(new Date());
      Logger.log("อัปเดตเวลาแก้ไขแถว " + row);
    }
  }
}

Installable Trigger (ต้องสร้างเอง — มีสิทธิ์มากกว่า)

// onEdit แบบ Installable — สามารถส่ง Email, เข้าถึง Service อื่นได้
function onEditInstallable(e) {
  var sheet = e.source.getActiveSheet();
  var range = e.range;
  
  // ถ้าเปลี่ยนสถานะเป็น "อนุมัติ" → ส่งแจ้งเตือน
  if (range.getColumn() === 5 && range.getValue() === "อนุมัติ") {
    var name = sheet.getRange(range.getRow(), 2).getValue();
    var email = sheet.getRange(range.getRow(), 4).getValue();
    
    MailApp.sendEmail(email, "อนุมัติแล้ว!", "คุณ " + name + " ได้รับการอนุมัติแล้วครับ");
    Logger.log("ส่งแจ้งเตือนอนุมัติให้ " + name);
  }
}

// สร้าง Installable onEdit Trigger
function setupEditTrigger() {
  ScriptApp.newTrigger("onEditInstallable")
    .forSpreadsheet(SpreadsheetApp.getActiveSpreadsheet())
    .onEdit()
    .create();
}

🗑️ จัดการ Trigger (ดู/ลบ)

// ดู Trigger ทั้งหมด
function listAllTriggers() {
  var triggers = ScriptApp.getProjectTriggers();
  Logger.log("จำนวน Trigger: " + triggers.length);
  
  for (var i = 0; i < triggers.length; i++) {
    var t = triggers[i];
    Logger.log((i+1) + ". " + t.getHandlerFunction() + " | " + t.getEventType());
  }
}

// ลบ Trigger ทั้งหมด
function deleteAllTriggers() {
  var triggers = ScriptApp.getProjectTriggers();
  for (var i = 0; i < triggers.length; i++) {
    ScriptApp.deleteTrigger(triggers[i]);
  }
  Logger.log("ลบ Trigger ทั้งหมด (" + triggers.length + " ตัว) สำเร็จ");
}

// ลบ Trigger เฉพาะฟังก์ชัน
function deleteTriggerByName(functionName) {
  var triggers = ScriptApp.getProjectTriggers();
  for (var i = 0; i < triggers.length; i++) {
    if (triggers[i].getHandlerFunction() === functionName) {
      ScriptApp.deleteTrigger(triggers[i]);
      Logger.log("ลบ Trigger: " + functionName);
    }
  }
}
⚠️ ข้อจำกัดของ Trigger
  • แต่ละ Script มี Trigger ได้สูงสุด 20 ตัว
  • Time-based จะทำงาน ภายใน ±15 นาที ของเวลาที่กำหนด
  • Simple Trigger (onOpen, onEdit) ส่ง Email หรือเข้าถึง Service อื่นไม่ได้
  • ต้องใช้ Installable Trigger ถ้าต้องการสิทธิ์เพิ่มเติม

🔬 Lab 7: สร้างระบบแจ้งเตือนอัตโนมัติ

เวลา: 25 นาที

โจทย์: สร้าง Trigger 3 แบบ

1

สร้างเมนูพิเศษเมื่อเปิด Sheet (onOpen)

2

ใส่ Timestamp อัตโนมัติเมื่อแก้ไข (onEdit)

3

ส่งรายงานสรุปทุกวัน (Time-based)

// ===== Lab 7: ระบบ Trigger ครบชุด =====

function onOpen() {
  SpreadsheetApp.getUi().createMenu("⚙️ ระบบ")
    .addItem("📊 ดูสรุป", "showSummary")
    .addItem("⏰ ตั้ง Trigger", "setupAllTriggers")
    .addItem("🗑️ ลบ Trigger ทั้งหมด", "deleteAllTriggers")
    .addToUi();
}

function onEdit(e) {
  var sheet = e.source.getActiveSheet();
  var row = e.range.getRow();
  var col = e.range.getColumn();
  
  // ใส่เวลาแก้ไขที่คอลัมน์สุดท้าย
  if (row > 1 && col < 6) {
    var lastCol = sheet.getLastColumn();
    sheet.getRange(row, lastCol).setValue(new Date());
  }
}

function showSummary() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var lastRow = sheet.getLastRow();
  SpreadsheetApp.getUi().alert(
    "📊 สรุป\n\nจำนวนข้อมูล: " + (lastRow - 1) + " รายการ\n" +
    "Sheet: " + sheet.getName()
  );
}

function sendDailyReport() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var lastRow = sheet.getLastRow();
  var message = "📊 รายงานประจำวัน\n";
  message += "วันที่: " + new Date().toLocaleDateString("th-TH") + "\n";
  message += "จำนวนข้อมูลทั้งหมด: " + (lastRow - 1) + " รายการ\n";
  Logger.log(message);
  // MailApp.sendEmail("your@email.com", "รายงานประจำวัน", message);
}

function setupAllTriggers() {
  // ลบ Trigger เดิมก่อน
  deleteAllTriggers();
  
  // สร้าง Daily Report เวลา 08:00
  ScriptApp.newTrigger("sendDailyReport")
    .timeBased()
    .atHour(8)
    .everyDays(1)
    .create();
  
  SpreadsheetApp.getUi().alert("ตั้ง Trigger สำเร็จ! 🎉");
}

💡 Tips

  • ลบ Trigger เดิมก่อนสร้างใหม่: ป้องกัน Trigger ซ้ำซ้อน
  • Simple vs Installable: Simple (onOpen, onEdit) ง่ายแต่สิทธิ์จำกัด, Installable สิทธิ์เต็มแต่ต้องสร้างเอง
  • ทดสอบด้วย Logger.log ก่อน: แล้วค่อยเปิด Email/Line จริง
  • ดู Trigger ที่หน้า Triggers: คลิก ⏰ ด้านซ้ายของ Script Editor
