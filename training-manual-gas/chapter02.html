บทที่ 2: Google Sheets เป็นฐานข้อมูล

เรียนรู้การใช้ SpreadsheetApp เพื่ออ่าน/เขียนข้อมูลใน Google Sheets อัตโนมัติ

🎯 วัตถุประสงค์การเรียนรู้

  • ใช้ SpreadsheetApp เข้าถึง Google Sheets ได้
  • อ่านข้อมูลจาก Sheet ด้วย getValues()
  • เขียนข้อมูลลง Sheet ด้วย setValues()
  • เข้าใจแนวคิด Range, Sheet, Spreadsheet

📚 ทำไมต้องใช้ Google Sheets เป็นฐานข้อมูล?

ในโลกของการเขียนโปรแกรม ปกติเราต้องใช้ฐานข้อมูล (Database) เช่น MySQL, PostgreSQL ซึ่งต้องติดตั้งและดูแลรักษา แต่สำหรับงานขนาดเล็ก-กลาง Google Sheets ทำหน้าที่เป็นฐานข้อมูลได้ดีมาก!

✅ ข้อดีของ Sheets เป็นฐานข้อมูล

  • มองเห็นข้อมูลทั้งหมด — ไม่ต้องเขียน Query เพื่อดูข้อมูล
  • แก้ไขง่าย — เจ้าหน้าที่แก้ข้อมูลใน Sheet ได้โดยตรง
  • แชร์ได้ — หลายคนใช้งานพร้อมกัน
  • ไม่ต้องติดตั้ง — ใช้เบราว์เซอร์เปิดได้เลย
  • เหมาะกับข้อมูลไม่เกิน 50,000 แถว

🏗️ โครงสร้างลำดับชั้น: Spreadsheet → Sheet → Range → Cell

// ลำดับชั้นของ Google Sheets
// 
// Spreadsheet (ไฟล์ .gsheet ทั้งไฟล์)
//   └── Sheet (แท็บ เช่น Sheet1, Sheet2)
//         └── Range (กลุ่มเซลล์ เช่น A1:C10)
//               └── Cell (เซลล์เดียว เช่น A1)

// เปรียบเทียบกับฐานข้อมูล:
// Spreadsheet = Database
// Sheet       = Table
// Row         = Record (แถว = 1 รายการข้อมูล)
// Column      = Field (คอลัมน์ = ประเภทข้อมูล)

📖 SpreadsheetApp — เข้าถึง Spreadsheet

// วิธีที่ 1: เปิด Spreadsheet ที่ Script ผูกอยู่
var ss = SpreadsheetApp.getActiveSpreadsheet();

// วิธีที่ 2: เปิดจาก URL
var ss = SpreadsheetApp.openByUrl("https://docs.google.com/spreadsheets/d/xxx/edit");

// วิธีที่ 3: เปิดจาก ID (แนะนำ)
var ss = SpreadsheetApp.openById("1abc123xyz...");

// เข้าถึง Sheet (แท็บ)
var sheet = ss.getSheetByName("Sheet1");        // เปิดตามชื่อ
var sheet = ss.getActiveSheet();                 // แท็บที่เปิดอยู่
var sheet = ss.getSheets()[0];                   // แท็บแรก

📥 อ่านข้อมูล (Read) — getValues()

function readData() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("ข้อมูลสมาชิก");
  
  // อ่านเซลล์เดียว
  var name = sheet.getRange("A2").getValue();
  Logger.log("ชื่อ: " + name);
  
  // อ่านหลายเซลล์ (ได้ Array 2 มิติ)
  var data = sheet.getRange("A2:C5").getValues();
  Logger.log(data);
  // ผลลัพธ์: [["ปัญญา","สุรินทร์",30], ["สมชาย","กรุงเทพ",25], ...]
  
  // อ่านทั้ง Sheet
  var allData = sheet.getDataRange().getValues();
  Logger.log("จำนวนแถว: " + allData.length);
  
  // วนลูปอ่านทีละแถว
  for (var i = 1; i < allData.length; i++) {  // เริ่มที่ 1 ข้าม header
    var row = allData[i];
    Logger.log("ชื่อ: " + row[0] + " | จังหวัด: " + row[1] + " | อายุ: " + row[2]);
  }
}
⚠️ สิ่งที่ต้องจำ:
  • getValue() = อ่านเซลล์เดียว → ได้ค่าเดียว
  • getValues() = อ่านหลายเซลล์ → ได้ Array 2 มิติ (แถว × คอลัมน์)
  • Array เริ่มนับจาก 0 → data[0][0] = เซลล์แรก
  • แถวแรก (index 0) มักเป็น Header → วนลูปเริ่มที่ i=1

📤 เขียนข้อมูล (Write) — setValues()

function writeData() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("ข้อมูลสมาชิก");
  
  // เขียนเซลล์เดียว
  sheet.getRange("A2").setValue("ปัญญา");
  
  // เขียนหลายเซลล์พร้อมกัน (ต้องเป็น Array 2 มิติ)
  var newData = [
    ["สมชาย", "กรุงเทพ", 25],
    ["สมหญิง", "เชียงใหม่", 28],
    ["สมศักดิ์", "สุรินทร์", 32]
  ];
  sheet.getRange("A2:C4").setValues(newData);
  
  // เพิ่มแถวใหม่ที่ท้าย Sheet
  sheet.appendRow(["คนใหม่", "ยโสธร", 22]);
  
  Logger.log("เขียนข้อมูลสำเร็จ!");
}

🔍 เทคนิคสำคัญ: getLastRow() และ getLastColumn()

function sheetInfo() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  
  // หาแถวสุดท้ายที่มีข้อมูล
  var lastRow = sheet.getLastRow();
  Logger.log("แถวสุดท้าย: " + lastRow);
  
  // หาคอลัมน์สุดท้ายที่มีข้อมูล
  var lastCol = sheet.getLastColumn();
  Logger.log("คอลัมน์สุดท้าย: " + lastCol);
  
  // อ่านข้อมูลทั้งหมด (ไม่รวม header)
  var dataRange = sheet.getRange(2, 1, lastRow - 1, lastCol);
  var data = dataRange.getValues();
  Logger.log("จำนวนรายการ: " + data.length);
}

🎨 จัดรูปแบบ Cell

function formatCells() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  
  // เปลี่ยนสีพื้นหลัง Header
  sheet.getRange("A1:D1").setBackground("#4285f4").setFontColor("#ffffff");
  
  // ตั้งความกว้างคอลัมน์
  sheet.setColumnWidth(1, 150);  // คอลัมน์ A กว้าง 150px
  
  // ตัวหนา
  sheet.getRange("A1:D1").setFontWeight("bold");
  
  // จัดกลาง
  sheet.getRange("A1:D1").setHorizontalAlignment("center");
}

🔬 Lab 4: อ่านข้อมูลและแสดงรายงาน

เวลา: 15 นาที

function lab4_readReport() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = sheet.getDataRange().getValues();
  
  Logger.log("===== รายงานสมาชิก =====");
  Logger.log("จำนวนสมาชิกทั้งหมด: " + (data.length - 1) + " คน");
  Logger.log("========================");
  
  for (var i = 1; i < data.length; i++) {
    Logger.log(i + ". " + data[i][1] + " (" + data[i][2] + ") โทร: " + data[i][3]);
  }
  
  // นับจำนวนสมาชิกแต่ละจังหวัด
  var provinces = {};
  for (var i = 1; i < data.length; i++) {
    var prov = data[i][2];
    provinces[prov] = (provinces[prov] || 0) + 1;
  }
  
  Logger.log("\n===== สถิติแยกจังหวัด =====");
  for (var prov in provinces) {
    Logger.log(prov + ": " + provinces[prov] + " คน");
  }
}

🎯 ผลที่คาดหวัง: เห็นรายชื่อสมาชิกและสถิติแยกจังหวัดใน Log

💡 Tips ที่ควรรู้

  • getDataRange() — อ่านเฉพาะส่วนที่มีข้อมูล (ไม่รวมเซลล์ว่าง)
  • appendRow() — เพิ่มแถวท้ายสุดอัตโนมัติ ไม่ต้องหา lastRow เอง
  • ระวัง setValues() vs setValue() — setValues ต้องเป็น Array 2 มิติเสมอ
  • ประหยัด Quota: อ่าน/เขียนทีเดียวทั้งก้อน ดีกว่าทำทีละเซลล์

📝 สรุปบทที่ 2

SpreadsheetApp จุดเริ่มต้นในการเข้าถึง Google Sheets ทั้งหมด
getValues() อ่านข้อมูลหลายเซลล์ → ได้ Array 2 มิติ
setValues() เขียนข้อมูลหลายเซลล์พร้อมกัน (ต้อง Array 2 มิติ)
appendRow() เพิ่มแถวใหม่ท้าย Sheet (Array 1 มิติ)
getLastRow() หาแถวสุดท้ายที่มีข้อมูล
