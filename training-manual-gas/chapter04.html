บทที่ 4: Docs AI + Form

สร้างเอกสารอัตโนมัติจาก Template + เชื่อมต่อ Google Forms

🎯 วัตถุประสงค์การเรียนรู้

  • สร้างเอกสาร Google Docs อัตโนมัติจาก Template
  • ใช้เทคนิค Find & Replace แทนที่ข้อความใน Template
  • เชื่อมต่อ Google Forms กับ Apps Script
  • ใช้ onFormSubmit Trigger ประมวลผลเมื่อมีคนส่งฟอร์ม

📄 สร้างเอกสารอัตโนมัติ (Document Automation)

สถานการณ์ที่พบบ่อย: ต้องสร้างเอกสารเหมือนกันหลายฉบับ แต่เปลี่ยนแค่ชื่อ วันที่ หมายเลข เช่น หนังสือรับรอง ใบประกาศ เกียรติบัตร

🔄 หลักการทำงาน

  1. สร้าง Template (ต้นแบบ) ใน Google Docs — ใส่ตัวแทนที่ เช่น {{ชื่อ}}
  2. GAS คัดลอก Template เป็นไฟล์ใหม่
  3. GAS แทนที่ {{ชื่อ}} ด้วยข้อมูลจริง
  4. ได้เอกสารพร้อมใช้งาน!

📝 ขั้นตอนที่ 1: สร้าง Template

เปิด Google Docs สร้างเอกสารใหม่ เขียนเนื้อหาแบบนี้:

หนังสือรับรองการเข้าร่วมอบรม

เลขที่: {{เลขที่}}
วันที่: {{วันที่}}

ขอรับรองว่า {{ชื่อ}} ตำแหน่ง {{ตำแหน่ง}}
สังกัด {{หน่วยงาน}}
ได้เข้าร่วมการอบรมหลักสูตร "{{หลักสูตร}}"
ระหว่างวันที่ {{วันเริ่ม}} ถึง {{วันสิ้นสุด}}

จึงขอรับรองมา ณ ที่นี้

⚙️ ขั้นตอนที่ 2: เขียนโค้ดสร้างเอกสาร

function createDocument(data) {
  // ID ของ Template (คัดลอกจาก URL ของ Google Docs)
  var templateId = "1abc123_YOUR_TEMPLATE_ID_HERE";
  
  // คัดลอก Template เป็นไฟล์ใหม่
  var template = DriveApp.getFileById(templateId);
  var newFile = template.makeCopy("หนังสือรับรอง - " + data.name);
  var newDoc = DocumentApp.openById(newFile.getId());
  var body = newDoc.getBody();
  
  // แทนที่ข้อความ
  body.replaceText("{{เลขที่}}", data.docNumber);
  body.replaceText("{{วันที่}}", data.date);
  body.replaceText("{{ชื่อ}}", data.name);
  body.replaceText("{{ตำแหน่ง}}", data.position);
  body.replaceText("{{หน่วยงาน}}", data.organization);
  body.replaceText("{{หลักสูตร}}", data.course);
  body.replaceText("{{วันเริ่ม}}", data.startDate);
  body.replaceText("{{วันสิ้นสุด}}", data.endDate);
  
  // บันทึกและปิด
  newDoc.saveAndClose();
  
  Logger.log("สร้างเอกสารสำเร็จ: " + newFile.getUrl());
  return newFile.getUrl();
}

📋 สร้างเอกสารจาก Sheet (หลายคนพร้อมกัน)

function createDocsFromSheet() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("ผู้เข้าอบรม");
  var data = sheet.getDataRange().getValues();
  var templateId = "1abc123_YOUR_TEMPLATE_ID_HERE";
  
  // วนลูปสร้างเอกสารทีละคน (ข้าม header)
  for (var i = 1; i < data.length; i++) {
    var row = data[i];
    var docData = {
      docNumber: "อบ." + String(i).padStart(3, "0") + "/2569",
      date: "24 กรกฎาคม 2569",
      name: row[1],        // คอลัมน์ B = ชื่อ
      position: row[2],    // คอลัมน์ C = ตำแหน่ง
      organization: row[3],// คอลัมน์ D = หน่วยงาน
      course: "การประยุกต์ใช้ปัญญาประดิษฐ์ในทุกมิติ เพื่อคนพิการ",
      startDate: "20 กรกฎาคม 2569",
      endDate: "24 กรกฎาคม 2569"
    };
    
    var url = createDocument(docData);
    // บันทึก URL ลงคอลัมน์ท้ายสุด
    sheet.getRange(i + 1, data[0].length + 1).setValue(url);
    
    Logger.log("สร้างเอกสาร " + i + "/" + (data.length - 1) + ": " + row[1]);
  }
  
  SpreadsheetApp.getUi().alert("สร้างเอกสารเสร็จ " + (data.length - 1) + " ฉบับ");
}

📝 Google Forms + Apps Script

Google Forms สามารถเชื่อมต่อกับ Apps Script ผ่าน onFormSubmit Trigger — ทุกครั้งที่มีคนส่งฟอร์ม GAS จะทำงานอัตโนมัติ

// ฟังก์ชันที่ทำงานเมื่อมีคนส่ง Form
function onFormSubmit(e) {
  // e.values = ข้อมูลที่กรอกในฟอร์ม (Array)
  // e.values[0] = Timestamp
  // e.values[1] = คำตอบข้อ 1
  // e.values[2] = คำตอบข้อ 2 ...
  
  var timestamp = e.values[0];
  var name = e.values[1];
  var email = e.values[2];
  var phone = e.values[3];
  
  Logger.log("ได้รับข้อมูลจากฟอร์ม: " + name + " (" + email + ")");
  
  // ส่งอีเมลยืนยัน
  MailApp.sendEmail(email, 
    "ยืนยันการลงทะเบียน", 
    "สวัสดีคุณ " + name + "\n\nลงทะเบียนสำเร็จแล้วครับ!\n\nขอบคุณครับ"
  );
  
  Logger.log("ส่งอีเมลยืนยันให้ " + email + " แล้ว");
}

🔧 วิธีตั้ง onFormSubmit Trigger

1

เปิด Google Sheets ที่เชื่อมกับ Form (Form responses)

2

ไปที่ Extensions → Apps Script

3

เขียนฟังก์ชัน onFormSubmit(e)

4

คลิกไอคอน ⏰ (Triggers) ด้านซ้าย → Add Trigger

5

ตั้งค่า: Function = onFormSubmit | Event = On form submit

// หรือตั้ง Trigger ด้วยโค้ด
function setupFormTrigger() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  ScriptApp.newTrigger("onFormSubmit")
    .forSpreadsheet(ss)
    .onFormSubmit()
    .create();
  Logger.log("ตั้ง Trigger สำเร็จ!");
}

function onFormSubmit_createDoc(e) {

🔬 Lab 6: สร้างเกียรติบัตรอัตโนมัติ

เวลา: 30 นาที

โจทย์: สร้างเกียรติบัตรการอบรมจาก Template อัตโนมัติ

1

สร้าง Google Docs Template ที่มีข้อความ {{ชื่อ}}, {{หลักสูตร}}, {{วันที่}}

2

คัดลอก ID ของ Template จาก URL

3

สร้าง Sheet "ผู้ผ่านอบรม" ใส่ข้อมูล 3 คน

4

เขียนโค้ดสร้างเกียรติบัตรให้ทุกคนจาก Sheet

function lab6_createCertificates() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("ผู้ผ่านอบรม");
  var data = sheet.getDataRange().getValues();
  var templateId = "YOUR_TEMPLATE_ID";  // ← ใส่ ID จริง
  
  for (var i = 1; i < data.length; i++) {
    var name = data[i][0];
    
    // คัดลอก Template
    var template = DriveApp.getFileById(templateId);
    var newFile = template.makeCopy("เกียรติบัตร - " + name);
    var doc = DocumentApp.openById(newFile.getId());
    var body = doc.getBody();
    
    // แทนที่ข้อความ
    body.replaceText("{{ชื่อ}}", name);
    body.replaceText("{{หลักสูตร}}", "การประยุกต์ใช้ AI เพื่อคนพิการ");
    body.replaceText("{{วันที่}}", "24 กรกฎาคม 2569");
    
    doc.saveAndClose();
    Logger.log("สร้างเกียรติบัตร: " + name + " → " + newFile.getUrl());
  }
  
  SpreadsheetApp.getUi().alert("สร้างเกียรติบัตรครบ " + (data.length-1) + " ฉบับ!");
}

💡 Tips

  • ตั้งชื่อ Placeholder ให้ unique: ใช้ {{ชื่อ}} ดีกว่า ชื่อ เพราะไม่ไปแทนที่คำอื่นโดยบังเอิญ
  • ย้ายไฟล์เข้า Folder: ใช้ DriveApp.getFolderById(folderId).addFile(newFile)
  • จำกัดการทำงาน: GAS รันได้ไม่เกิน 6 นาทีต่อครั้ง (ถ้าทำ 100+ ไฟล์ ต้องแบ่งรอบ)
  • Form Trigger: e.values เป็น Array เรียงตามลำดับคำถามในฟอร์ม
