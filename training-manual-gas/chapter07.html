บทที่ 7: HTML Service + Bootstrap

สร้างหน้าเว็บสวยงามด้วย GAS — พร้อมเชื่อมต่อ Backend ด้วย google.script.run

🎯 วัตถุประสงค์การเรียนรู้

  • สร้างหน้าเว็บ (HTML) ด้วย HtmlService
  • ใช้ Bootstrap CDN ออกแบบ UI สวยงาม
  • ใช้ google.script.run เรียกฟังก์ชัน Backend
  • สร้างระบบ Frontend-Backend ที่สมบูรณ์

📚 HtmlService คืออะไร?

HtmlService ให้เราสร้างหน้าเว็บ (HTML) ที่ทำงานบน GAS ได้ — เหมือนมีเว็บไซต์ฟรีที่ไม่ต้องเช่า Hosting!

🏗️ โครงสร้างของ GAS Web App

  • Code.gs — Backend (ฟังก์ชันประมวลผล, อ่าน/เขียน Sheet)
  • index.html — Frontend (หน้าเว็บที่ผู้ใช้เห็น)
  • google.script.run — สะพานเชื่อม Frontend ↔ Backend

<body>
  <div class="container">
    <h1 class="text-primary">🎉 ระบบจัดการสมาชิก</h1>
    <p>สร้างด้วย Google Apps Script + Bootstrap</p>
    
    <div class="card">
      <div class="card-body">
        <h5 class="card-title">เพิ่มสมาชิกใหม่</h5>
        <div class="mb-3">
          <input type="text" id="txtName" class="form-control" placeholder="ชื่อ-นามสกุล">
        </div>
        <div class="mb-3">
          <input type="text" id="txtPhone" class="form-control" placeholder="โทรศัพท์">
        </div>
        <button onclick="addMember()" class="btn btn-primary">เพิ่มสมาชิก</button>
        <div id="result" class="mt-3"></div>
      </div>
    </div>

  <script>
    function addMember() {
      var name = document.getElementById('txtName').value;
      var phone = document.getElementById('txtPhone').value;
      
      if (!name || !phone) {
        document.getElementById('result').innerHTML = 
          '<div class="alert alert-warning">กรุณากรอกข้อมูลให้ครบ</div>';
        return;
      }
      
      // เรียกฟังก์ชัน Backend
      google.script.run
        .withSuccessHandler(function(response) {
          document.getElementById('result').innerHTML = 
            '<div class="alert alert-success">' + response.message + '</div>';
          document.getElementById('txtName').value = '';
          document.getElementById('txtPhone').value = '';
        })
        .withFailureHandler(function(error) {
          document.getElementById('result').innerHTML = 
            '<div class="alert alert-danger">เกิดข้อผิดพลาด: ' + error + '</div>';
        })
        .addMemberFromWeb(name, phone);
    }
  </script>
</body>
</html>

ขั้นตอนที่ 2: เขียน Backend (Code.gs)

// แสดงหน้าเว็บ
function doGet() {
  return HtmlService.createHtmlOutputFromFile("index")
    .setTitle("ระบบจัดการสมาชิก")
    .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
}

// ฟังก์ชันที่ Frontend เรียก
function addMemberFromWeb(name, phone) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("สมาชิก");
  var id = sheet.getLastRow();
  sheet.appendRow([id, name, phone, new Date()]);
  
  return { success: true, message: "เพิ่ม " + name + " สำเร็จ (ID: " + id + ")" };
}

// ดึงข้อมูลสมาชิกทั้งหมด
function getMembersForWeb() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("สมาชิก");
  var data = sheet.getDataRange().getValues();
  var members = [];
  
  for (var i = 1; i < data.length; i++) {
    members.push({
      id: data[i][0],
      name: data[i][1],
      phone: data[i][2],
      date: data[i][3]
    });
  }
  return members;
}

🔗 google.script.run — เชื่อม Frontend กับ Backend

// รูปแบบการใช้งาน
google.script.run
  .withSuccessHandler(function(result) {
    // ทำงานเมื่อสำเร็จ — result คือค่าที่ Backend return
    console.log("สำเร็จ:", result);
  })
  .withFailureHandler(function(error) {
    // ทำงานเมื่อเกิดข้อผิดพลาด
    console.log("Error:", error);
  })
  .ชื่อฟังก์ชันใน_Code_gs(parameter1, parameter2);

// ตัวอย่าง: โหลดข้อมูลมาแสดง
function loadMembers() {
  document.getElementById('loading').style.display = 'block';
  
  google.script.run
    .withSuccessHandler(function(members) {
      var html = '';
      for (var i = 0; i < members.length; i++) {
        html += '<tr>';
        html += '<td>' + members[i].id + '</td>';
        html += '<td>' + members[i].name + '</td>';
        html += '<td>' + members[i].phone + '</td>';
        html += '</tr>';
      }
      document.getElementById('memberTable').innerHTML = html;
      document.getElementById('loading').style.display = 'none';
    })
    .getMembersForWeb();
}

<!-- index.html -->
<!DOCTYPE html>
<html>
<head>
  <base target="_top">
  <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/css/bootstrap.min.css" rel="stylesheet">
</head>
<body>
  <nav class="navbar navbar-dark bg-primary">
    <div class="container">
      <span class="navbar-brand">📋 ระบบลงทะเบียน</span>
    </div>
  </nav>

  <div class="container mt-4">
    <!-- ตารางข้อมูล -->
    <div class="card">
      <div class="card-header d-flex justify-content-between">
        <h5>รายชื่อสมาชิก</h5>
        <button class="btn btn-success btn-sm" onclick="showAddForm()">+ เพิ่ม</button>
      </div>
      <div class="card-body">
        <div id="loading" class="text-center">กำลังโหลด...</div>
        <table class="table table-striped" id="dataTable" style="display:none">
          <thead>
            <tr><th>ID</th><th>ชื่อ</th><th>โทร</th><th>จัดการ</th></tr>
          </thead>
          <tbody id="memberTable"></tbody>
        </table>
      </div>
    </div>
  </div>

  <script>
    // โหลดข้อมูลเมื่อเปิดหน้า
    window.onload = function() { loadMembers(); };
    
    function loadMembers() {
      google.script.run
        .withSuccessHandler(function(data) {
          var html = '';
          for (var i = 0; i < data.length; i++) {
            html += '<tr>';
            html += '<td>' + data[i].id + '</td>';
            html += '<td>' + data[i].name + '</td>';
            html += '<td>' + data[i].phone + '</td>';
            html += '<td><button class="btn btn-danger btn-sm" onclick="deleteMember(' + data[i].id + ')">ลบ</button></td>';
            html += '</tr>';
          }
          document.getElementById('memberTable').innerHTML = html;
          document.getElementById('loading').style.display = 'none';
          document.getElementById('dataTable').style.display = 'table';
        })
        .getMembersForWeb();
    }
    
    function deleteMember(id) {
      if (!confirm('ยืนยันลบ ID: ' + id + ' ?')) return;
      google.script.run
        .withSuccessHandler(function() { loadMembers(); })
        .deleteMemberFromWeb(id);
    }
  </script>
</body>
</html>

💡 Tips

  • base target="_top" — ใส่ทุกหน้า HTML เพื่อให้ลิงก์ทำงานถูกต้อง
  • google.script.run ช้า: ใช้เวลา 1-3 วินาที — ควรแสดง loading
  • ไม่ return HTML Object ได้: google.script.run คืนค่าได้เฉพาะ String, Number, Array, Object ธรรมดา
  • ใช้ Bootstrap CDN: ไม่ต้องดาวน์โหลด ใส่ link ใน head ได้เลย
