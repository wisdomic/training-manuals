📖 Google Apps Script บทที่ 6: doGet / doPost

บทที่ 6: doGet / doPost — Web App

สร้าง Web App ที่รับ-ส่งข้อมูลจากภายนอก — เชื่อมต่อ LINE, Webhook

🎯 วัตถุประสงค์การเรียนรู้

  • เข้าใจหลักการทำงานของ doGet() และ doPost()
  • สร้าง Web App ที่ตอบกลับเป็น JSON ได้
  • รับข้อมูลจากภายนอกผ่าน URL Parameters
  • เชื่อมต่อกับ LINE Messaging API (Webhook)

📚 doGet / doPost คืออะไร?

เมื่อ Deploy GAS เป็น Web App จะได้ URL ที่คนอื่นเข้าถึงได้ โดย:

  • doGet(e) — ทำงานเมื่อมีคน เปิด URL (เหมือนเปิดเว็บ)
  • doPost(e) — ทำงานเมื่อมีระบบอื่น ส่งข้อมูลมา (เหมือนส่งฟอร์ม)

🔄 เปรียบเทียบ GET vs POST

GET POST
ใช้ดึงข้อมูล (อ่าน) ใช้ส่งข้อมูล (เขียน)
ข้อมูลอยู่ใน URL (?key=value) ข้อมูลอยู่ใน Body (ซ่อน)
เปิดจากเบราว์เซอร์ได้ ต้องใช้โปรแกรมส่ง
ตัวอย่าง: ดูข้อมูลสมาชิก ตัวอย่าง: LINE Webhook, ส่งฟอร์ม

📤 doGet — ตอบกลับ JSON

// doGet ส่งข้อมูลกลับเป็น JSON
function doGet(e) {
  // รับ parameter จาก URL
  // เช่น URL?action=list&province=สุรินทร์
  var action = e.parameter.action || "hello";
  
  var result = {};
  
  if (action === "hello") {
    result = { status: "ok", message: "สวัสดีจาก GAS Web App!" };
    
  } else if (action === "list") {
    // อ่านข้อมูลจาก Sheet
    var sheet = SpreadsheetApp.openById("SHEET_ID").getSheetByName("สมาชิก");
    var data = sheet.getDataRange().getValues();
    var members = [];
    
    for (var i = 1; i < data.length; i++) {
      members.push({
        id: data[i][0],
        name: data[i][1],
        province: data[i][2]
      });
    }
    result = { status: "ok", data: members, count: members.length };
    
  } else if (action === "search") {
    var keyword = e.parameter.keyword || "";
    // ค้นหา...
    result = { status: "ok", keyword: keyword, data: [] };
    
  } else {
    result = { status: "error", message: "ไม่รู้จัก action: " + action };
  }
  
  // ส่ง JSON กลับ
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

📥 doPost — รับข้อมูลจากภายนอก

function doPost(e) {
  try {
    // รับข้อมูล JSON จาก Body
    var body = JSON.parse(e.postData.contents);
    
    var action = body.action;
    var result = {};
    
    if (action === "register") {
      // เพิ่มสมาชิกใหม่
      var sheet = SpreadsheetApp.openById("SHEET_ID").getSheetByName("สมาชิก");
      sheet.appendRow([
        new Date(),
        body.name,
        body.province,
        body.phone,
        body.email
      ]);
      result = { status: "ok", message: "ลงทะเบียนสำเร็จ: " + body.name };
      
    } else if (action === "update") {
      // อัปเดตข้อมูล
      result = { status: "ok", message: "อัปเดตสำเร็จ" };
      
    } else {
      result = { status: "error", message: "ไม่รู้จัก action" };
    }
    
    return ContentService.createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({
      status: "error",
      message: err.toString()
    })).setMimeType(ContentService.MimeType.JSON);
  }
}

📱 เชื่อมต่อ LINE Messaging API

LINE Webhook ส่งข้อมูลมาทาง POST เมื่อมีคนส่งข้อความหา Bot:

var LINE_TOKEN = "YOUR_CHANNEL_ACCESS_TOKEN";

function doPost(e) {
  var body = JSON.parse(e.postData.contents);
  var events = body.events;
  
  for (var i = 0; i < events.length; i++) {
    var event = events[i];
    
    if (event.type === "message" && event.message.type === "text") {
      var userMessage = event.message.text;
      var replyToken = event.replyToken;
      var userId = event.source.userId;
      
      // บันทึกข้อความลง Sheet
      var sheet = SpreadsheetApp.openById("SHEET_ID").getSheetByName("LINE_LOG");
      sheet.appendRow([new Date(), userId, userMessage]);
      
      // ตอบกลับ
      var replyMessage = "ได้รับข้อความ: " + userMessage;
      replyToLine(replyToken, replyMessage);
    }
  }
  
  return ContentService.createTextOutput("OK");
}

function replyToLine(replyToken, message) {
  var url = "https://api.line.me/v2/bot/message/reply";
  var payload = {
    replyToken: replyToken,
    messages: [{ type: "text", text: message }]
  };
  
  UrlFetchApp.fetch(url, {
    method: "post",
    headers: {
      "Content-Type": "application/json",
      "Authorization": "Bearer " + LINE_TOKEN
    },
    payload: JSON.stringify(payload)
  });
}

// Push Message (ส่งข้อความหาคนโดยตรง)
function pushMessage(userId, message) {
  var url = "https://api.line.me/v2/bot/message/push";
  var payload = {
    to: userId,
    messages: [{ type: "text", text: message }]
  };
  
  UrlFetchApp.fetch(url, {
    method: "post",
    headers: {
      "Content-Type": "application/json",
      "Authorization": "Bearer " + LINE_TOKEN
    },
    payload: JSON.stringify(payload)
  });
}

โจทย์: สร้าง Web App ที่มี API ดึงข้อมูลสมาชิก

function doGet(e) {
  var action = e.parameter.action || "info";
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("สมาชิก");
  var result = {};
  
  switch(action) {
    case "info":
      result = {
        status: "ok",
        app: "ระบบสมาชิก",
        version: "1.0",
        author: "นายปัญญา ผาธรรม"
      };
      break;
      
    case "list":
      var data = sheet.getDataRange().getValues();
      var members = [];
      for (var i = 1; i < data.length; i++) {
        members.push({id: data[i][0], name: data[i][1], province: data[i][2]});
      }
      result = { status: "ok", data: members, count: members.length };
      break;
      
    case "get":
      var id = e.parameter.id;
      var data = sheet.getDataRange().getValues();
      var found = null;
      for (var i = 1; i < data.length; i++) {
        if (String(data[i][0]) === String(id)) {
          found = {id: data[i][0], name: data[i][1], province: data[i][2], phone: data[i][3]};
          break;
        }
      }
      result = found ? {status: "ok", data: found} : {status: "error", message: "ไม่พบ ID: " + id};
      break;
      
    case "count":
      var lastRow = sheet.getLastRow();
      result = { status: "ok", count: lastRow - 1 };
      break;
      
    default:
      result = { status: "error", message: "ไม่รู้จัก action: " + action };
  }
  
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

💡 Tips

  • ทดสอบ doPost: ใช้ Postman หรือ curl ส่ง request
  • ความปลอดภัย: ตรวจสอบ token/password ก่อนประมวลผล
  • Error Handling: ใส่ try-catch เสมอ เพื่อป้องกัน crash
  • LINE Webhook URL: ต้อง Deploy ก่อน แล้วนำ URL ไปใส่ใน LINE Developers Console
  • CORS: GAS Web App ไม่มีปัญหา CORS เพราะ Google จัดการให้
