function sendSimpleEmail() {

}

📎 แนบไฟล์จาก Google Drive

function sendEmailWithAttachment() {
  // แนบไฟล์จาก Drive (ใช้ File ID)
  var file = DriveApp.getFileById("FILE_ID_HERE");
  
  MailApp.sendEmail({
    to: "recipient@example.com",
    subject: "เอกสารแนบ",
    body: "กรุณาดูเอกสารแนบ",
    attachments: [file.getAs(MimeType.PDF)]  // แนบเป็น PDF
  });
}

// แนบหลายไฟล์
function sendEmailMultipleAttachments() {
  var folder = DriveApp.getFolderById("FOLDER_ID");
  var files = folder.getFiles();
  var attachments = [];
  
  while (files.hasNext()) {
    var file = files.next();
    attachments.push(file.getBlob());
  }
  
  MailApp.sendEmail({
    to: "recipient@example.com",
    subject: "ไฟล์ทั้งหมดในโฟลเดอร์",
    body: "แนบ " + attachments.length + " ไฟล์",
    attachments: attachments
  });
}

// สร้าง PDF จาก Spreadsheet แล้วแนบ
function sendSheetAsPDF() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var pdf = ss.getAs("application/pdf");
  pdf.setName("รายงาน_" + new Date().toLocaleDateString("th-TH") + ".pdf");
  
  MailApp.sendEmail({
    to: "recipient@example.com",
    subject: "รายงานประจำวัน (PDF)",
    body: "กรุณาดูรายงานแนบ",
    attachments: [pdf]
  });
}

⚡ ส่งอีเมลตามเงื่อนไข

// ส่งแจ้งเตือนเมื่อมีข้อมูลใหม่ (ใช้กับ Trigger)
function checkAndNotify() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("ลงทะเบียน");
  var data = sheet.getDataRange().getValues();
  
  var today = new Date();
  today.setHours(0, 0, 0, 0);
  var newRegistrations = [];
  
  for (var i = 1; i < data.length; i++) {
    var regDate = new Date(data[i][4]);  // คอลัมน์วันที่
    regDate.setHours(0, 0, 0, 0);
    
    if (regDate.getTime() === today.getTime()) {
      newRegistrations.push(data[i]);
    }
  }
  
  if (newRegistrations.length > 0) {
    var message = "📋 สรุปการลงทะเบียนวันนี้\n\n";
    message += "จำนวนใหม่: " + newRegistrations.length + " คน\n\n";
    
    for (var i = 0; i < newRegistrations.length; i++) {
      message += (i + 1) + ". " + newRegistrations[i][1] + " (" + newRegistrations[i][2] + ")\n";
    }
    
    MailApp.sendEmail("admin@example.com", "📊 รายงานลงทะเบียนวันนี้ (" + newRegistrations.length + " คน)", message);
    Logger.log("ส่งรายงาน: " + newRegistrations.length + " คนใหม่");
  }
}

📨 ส่ง Email จำนวนมาก (Bulk)

function sendBulkEmail() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("ผู้เข้าอบรม");
  var data = sheet.getDataRange().getValues();
  
  var quota = MailApp.getRemainingDailyQuota();
  var sent = 0;
  
  for (var i = 1; i < data.length; i++) {
    if (sent >= quota - 10) {  // เหลือ 10 ฉบับไว้ใช้อื่น
      Logger.log("Quota ใกล้หมด! หยุดที่ " + sent + " ฉบับ");
      break;
    }
    
    var email = data[i][3];  // คอลัมน์ Email
    var name = data[i][1];
    
    if (!email || email.indexOf("@") === -1) continue;  // ข้ามถ้า email ไม่ถูกต้อง
    
    try {
      MailApp.sendEmail({
        to: email,
        subject: "📌 แจ้งกำหนดการอบรม",
        body: "สวัสดีคุณ " + name + "\n\nแจ้งเพื่อทราบ...\n\nขอบคุณครับ",
        name: "ระบบแจ้งเตือนอัตโนมัติ"
      });
      sent++;
      
      // บันทึกสถานะส่งแล้ว
      sheet.getRange(i + 1, 6).setValue("ส่งแล้ว " + new Date().toLocaleString("th-TH"));
      
    } catch (err) {
      sheet.getRange(i + 1, 6).setValue("ส่งไม่สำเร็จ: " + err);
    }
    
    // หน่วงเวลา 1 วินาที ป้องกัน rate limit
    Utilities.sleep(1000);
  }
  
  Logger.log("ส่งอีเมลสำเร็จ " + sent + " ฉบับ (เหลือ Quota: " + (quota - sent) + ")");
}

💡 Tips

  • Quota: บัญชี Gmail ฟรีส่งได้ 100 ฉบับ/วัน | Google Workspace ส่งได้ 1,500 ฉบับ/วัน
  • ตรวจ Quota: ใช้ MailApp.getRemainingDailyQuota() ก่อนส่ง bulk
  • HTML Email: ใช้ inline style เท่านั้น (CSS class ใช้ไม่ได้ใน email)
  • หน่วงเวลา: Utilities.sleep(1000) ป้องกัน rate limit เมื่อส่งหลายฉบับ
  • ทดสอบก่อน: ส่งให้ตัวเองก่อนเสมอ ก่อนส่ง bulk
